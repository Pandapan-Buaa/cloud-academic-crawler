            InputStream stream = new ClassPathResource("wos_su.txt").getInputStream();
            BufferedReader bf = new BufferedReader(new InputStreamReader(stream, "utf-8"));
            String line;
            List<String> lines;
            while ((line = bf.readLine()) != null) {
                if (StringUtil.isEmpty(line)) {
                    continue;
                }
                if (!line.contains("|")) {
                    continue;
                }
                lines = StringUtil.split(line, '|');
                if (lines.size() != 3) {
                    continue;
                }
                WOS_SU_MAP.put(lines.get(1).trim(), lines.get(2).trim());
            }
        } catch (Exception e) {

        } finally {
            log.info("init ============================== wos_su size : {}", WOS_SU_MAP.size());
